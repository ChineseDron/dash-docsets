DirectoryIterator::getExtension Example

<?php
$dir = new DirectoryIterator(dirname(__FILE__));
foreach ($dir as $fileinfo) {
    if (!$fileinfo->isDot()) {
        echo $fileinfo->getExtension() . "\n";
    }
}
?>
